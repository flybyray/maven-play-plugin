182     private String readContent( InputStream is )
183         throws IOException
184     {
185         StringBuffer buf = new StringBuffer();
186         InputStreamReader r = new InputStreamReader( is, "UTF-8" );
187         BufferedReader br = new BufferedReader( r );
188         try
189         {
190             String line = br.readLine();
191             while ( line != null )
192             {
193                 buf.append( line ).append( '\n' );
194                 line = br.readLine();
195             }
196         }
197         finally
198         {
199             br.close();
200         }
201         return buf.toString();
202     }

252                         new StoreStep( storedVars, new VoidSeleniumCommand( storedVars, commandProcessor, command,

309                     }
310                 }

313                     String assertWhat = command.substring( "assert".length() );

315                     {
316                         String innerCmd = "is" + assertWhat.replace( "NotPresent", "Present" );
317                         cmd =
318                             new AssertFalseStep( new BooleanSeleniumCommand( storedVars, commandProcessor, innerCmd,

320                     }
321                     else if ( assertWhat.startsWith( "Not" ) )
322                     {
323                         String innerCmd = assertWhat.substring( "Not".length() );

359                                 new AssertEqualsStep( new StringSeleniumCommand( storedVars, commandProcessor, "get"

364                 else if ( command.startsWith( "waitFor" ) )

374                     else
375                     {
376                         if ( waitForWhat.endsWith( "NotPresent" ) )

378                             String innerCmd = waitForWhat.replace( "NotPresent", "Present" );

380                                 new WaitForFalseStep( new BooleanSeleniumCommand( storedVars, commandProcessor, "is"

399                                 cmd =

410                                 param2 = param1; // value to compare with
411                                 param1 = ""; // parameterless command

413                             if ( isBooleanCommand( innerCmd ) )
414                             {
415                                 cmd =
416                                     new WaitForTrueStep( new BooleanSeleniumCommand( storedVars, commandProcessor, "is"

418                             }

420                             {
421                                 cmd =

424                                                            param2 );

428                 }

432                         new CommandStep(
433                                          new VoidSeleniumCommand( storedVars, commandProcessor, command, param1, param2 ) );
434                 }
435             }

437             // System.out.println( cmd.toString() );
438         }

448                 || "ElementPresent".equals( command ) || "Ordered".equals( command )

463     private void executeTestSteps( List<Step> steps )
464         throws Exception
465     {
466         StringBuffer testTraceBuf = new StringBuffer();
467         StringBuffer verificationFailuresBuf = new StringBuffer();
468         // dumpTest(steps);
469         // int stepsCnt = steps.size();
470         // int indent = String.valueOf(stepsCnt).length();
471         int indent = 4;
472         int line = 0;
473         for ( Step step : steps )
474         {
475             line++;
476             try
477             {
478                 step.execute();
479                 String logLine = dumpTestStep( line, indent, step );
480                 if ( traceTest )
481                 {
482                     System.out.println( logLine );
483                 }
484                 else
485                 {
486                     testTraceBuf.append( logLine ).append( '\n' );
487                 }
488             }
489             catch ( VerificationError e )
490             {
491                 String msg = "verification failure: " + e.getMessage();
492                 String logLine = dumpTestStep( line, indent, step, msg );
493                 if ( traceTest )
494                 {
495                     System.out.println( logLine );
496                 }
497                 else
498                 {
499                     testTraceBuf.append( logLine ).append( '\n' );
500                 }
501                 verificationFailuresBuf.append( '\n' ).append( e.getMessage() );
502             }
503             catch ( AssertionError e )
504             {
505                 String msg = "assertion failure: " + e.getMessage();
506                 String logLine = dumpTestStep( line, indent, step, msg );
507                 if ( traceTest )
508                 {
509                     System.out.println( logLine );
510                 }
511                 else
512                 {
513                     // buf.append( logLine ).append( '\n' );
514                     testTraceBuf.append( logLine );
515                     System.out.println( testTraceBuf.toString() );
516                 }
517                 throw e;
518             }
519             catch ( Error e )//TODO-should I handle errors?
520             {
521                 String msg = "error: " + e.getMessage();
522                 String logLine = dumpTestStep( line, indent, step, msg );
523                 if ( traceTest )
524                 {
525                     System.out.println( logLine );
526                 }
527                 else
528                 {
529                     // buf.append( logLine ).append( '\n' );
530                     testTraceBuf.append( logLine );
531                     System.out.println( testTraceBuf.toString() );
532                 }
533                 throw e;
534             }
535             catch ( RuntimeException e )
536             {
537                 String msg = "runtime exception: " + e.getMessage();
538                 String logLine = dumpTestStep( line, indent, step, msg );
539                 if ( traceTest )
540                 {
541                     System.out.println( logLine );
542                 }
543                 else
544                 {
545                     // buf.append( logLine ).append( '\n' );
546                     testTraceBuf.append( logLine );
547                     System.out.println( testTraceBuf.toString() );
548                 }
549                 throw e;
550             }
551         }
552 
553         if ( verificationFailuresBuf.length() > 0 )
554         {
555             if ( !traceTest )
556             {
557                 System.out.println( testTraceBuf.toString() );
558             }
559             Assert.fail( "There are verification failures:" + verificationFailuresBuf.toString() );
560         }
561     }

579     {
580         String strLine = String.valueOf( line );

582         {
583             strLine = " " + strLine;// TODO improve
